	//כפתור שמור פרוייקט
	function saveProject()
	{
		Add_Get_Tickets.method ="post";
		var flagsave=true;
		
			if( document.Add_Get_Tickets.selProjects.value=="-1")
			{
				swal({   title: "הודעה!",   text: "הוצאה מיעד תתאפשר למנהל בלבד",   type: "error", timer: 5000,   confirmButtonText: "OK" });
				flagsave=false;
			}
		
		//Date comparison
		var str1 = document.getElementById("txtHiddenEndDate").value;
		var str2 = document.getElementById("txtDateEnd").value;
		var dt1  = parseInt(str1.substring(0,2),10);
		var mon1 = parseInt(str1.substring(3,5),10);
		var yr1  = parseInt(str1.substring(6,10),10);
		var dt2  = parseInt(str2.substring(0,2),10);
		var mon2 = parseInt(str2.substring(3,5),10);
		var yr2  = parseInt(str2.substring(6,10),10);
		var date1 = new Date(yr1, mon1, dt1);
		var date2 = new Date(yr2, mon2, dt2);

		if(document.Add_Get_Tickets.txtDateEnd.value != '')
		{
			if(date1 < date2)
			{
				swal({   title: "הודעה!",   text: "לא ניתן להגדיר תאריך סיום למשימה גדול מתאריך סיום היעד : " + date1.getDate()+"/"+date1.getMonth()+"/"+date1.getFullYear(),   type: "error", timer: 5000,   confirmButtonText: "OK" });
				flagsave=false;
			}
		}

		if(0==1)
		{
			var ticket_startdate = document.getElementById("txtDateStart").value;
			var ticket_enddate = document.getElementById("txtDateEnd").value;
			if ((flagsave==true) && (ticket_startdate != "13/05/2025" || ticket_enddate != "13/05/2025"))
			{
				var descr = prompt("נא להקיש סיבה לשינוי, זו תצורף לבקשתך");
				if (descr != null && descr != "")
					document.getElementById('txtProjChangeReason').value = descr;
				else
					flagsave=false;
			}
		}
		if(flagsave==true)
		{
			Add_Get_Tickets.action="Add_Get_Tickets.asp?recid=2152924&caregiver=69566&pr=1";
			Add_Get_Tickets.submit();
		}

	}

	function preMoveTicket(flagPersonal)
	{
		$.ajax({
				url:"add_get_tickets.asp?getlasttipul=1&recid=2152924",
				success:function(data) 
				{
					// alert (6535119+"::"+data);
					if ((6535119==data)||(data==-1)||(6535119==-1))
					{
						MoveTicket(flagPersonal);
						return false;
					}
					else 
					{ 
						if (data!=-1) {
							swal({title: "הודעה!", text: "שים לב, הפניה עודכנה מאז טעינת המסך, יש לבצע רענון", type: "warning", showCancelButton: false, confirmButtonColor: "#DD6B55", confirmButtonText: "OK ", closeOnConfirm: false},
								function(isConfirm)
								{
									if (isConfirm)
									{
										location.reload();
									}
								}
							);
							return false
						}
					}
				},
				error:function(data){
					alert("Error In Ajax Query Please Try Again!");
					return false;
					}
			});	
	}

	//כפתור העברת פנייה
	function MoveTicket(flagPersonal){

		if(0 == 1 && flagPersonal == "False")
		{
		  var strconfirm = confirm("אם מעבירים  פנייה לצוות אחר נא לעדכן יעד מחדש");
		  if (strconfirm == true)
			 {
				Add_Get_Tickets.cmbMove.disabled = true;
				Add_Get_Tickets.method="post";
				Add_Get_Tickets.action="Move_Tickets.asp?recid=2152924&LTCS=6535119&caregiver=69566&groups=2152924&teams=2&pr=0&flagPersonal=" + flagPersonal;
				Add_Get_Tickets.submit();
			 }
		}
		else if(0 == 1 && flagPersonal == "True")
		{
			Add_Get_Tickets.cmbMove.disabled = true;
			Add_Get_Tickets.method="post";
			Add_Get_Tickets.action="Move_Tickets.asp?recid=2152924&LTCS=6535119&flagPersonal=" + flagPersonal;
			Add_Get_Tickets.submit();
		}
		else
		{
			Add_Get_Tickets.cmbMove.disabled = true;
			Add_Get_Tickets.method="post";
			Add_Get_Tickets.action="Move_Tickets.asp?recid=2152924&LTCS=6535119&caregiver=69566&groups=2152924&teams=2&pr=0&flagPersonal=" + flagPersonal;
			Add_Get_Tickets.submit();
		}
	}
